Now we can compute the REM. For this we’ll need to: 1. Calculate the elevation of the river along its path (from the DEM data) 2. Interpolate this to create a 2-D elevation raster 3. Subtract this interpolated elevation raster from the DEM

import xarray as xr
import numpy as np
from scipy.spatial import KDTree
# _gdf_river = gdf_river.to_crs("EPSG:32610")
river_geom = gdf_river.to_crs("EPSG:32610").iloc[0].geometry
river_geom = shapely.line_merge(river_geom)
import scipy#.interpolate
x, y = shapely.get_coordinates(river_geom).T
konts = np.hypot(np.diff(x), np.diff(y)).cumsum()
konts = np.insert(konts, 0, 0)
konts /= konts[-1]

smoothing = None
spl_x = scipy.interpolate.UnivariateSpline(konts, x, k=3, s=smoothing)
spl_y = scipy.interpolate.UnivariateSpline(konts, y, k=3, s=smoothing)
_npts = 5 * len(x)
konts = np.linspace(0, 1, _npts)
river_geom_smooth = shapely.LineString(np.c_[spl_x(konts), spl_y(konts)])
xs, ys = river_geom_smooth.xy
xs, ys = xr.DataArray(xs, dims='z'), xr.DataArray(ys, dims='z')

# Take nearest elevation to those coordinates
river_elevation = dem.interp(x=xs, y=ys, method='linear').dropna(dim='z')

To mitigate artifacts arising from the river geometry and DEM not matching (due to natural change over time), we can ensure that the river elevation decreases or stays constant between pixels

river_elevation.plot(figsize=(4, 2));
# Ensure the array is non-increasing
river_elevation.values = np.minimum.accumulate(river_elevation.values)
river_elevation.plot(figsize=(4, 2));
# Interpolate the river elevation using Inverse Distance Weighted interpolation
river_elevation_coords = np.column_stack((river_elevation.x, river_elevation.y))

kdt = KDTree(river_elevation_coords)
dem_grid = np.dstack(np.meshgrid(dem.x, dem.y)).reshape(-1, 2)
k = 50
distances, indices = kdt.query(dem_grid, k=k)
weights = np.reciprocal(distances)
weights = weights / weights.sum(axis=1, keepdims=True)
river_elevation_interpolated = weights * river_elevation.to_numpy()[indices]
river_elevation_interpolated = river_elevation_interpolated.sum(axis=1).reshape((dem.sizes["y"], dem.sizes["x"]))
river_elevation_interpolated = xr.DataArray(river_elevation_interpolated, dims=("y", "x"), coords={"x": dem.x, "y": dem.y})
river_elevation_interpolated.plot();
